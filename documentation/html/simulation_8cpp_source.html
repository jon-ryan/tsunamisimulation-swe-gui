130 void simulation::computeTimestep(bool& error_happened) {

132  #pragma omp parallel for schedule(static) default(none)
133  for (std::size_t x = 0; x < num_cells[0]; ++x) {
134  // Height
135  h[x] = h[num_cells[0] + x];
136  h[(num_cells[1] + 1) * num_cells[0] + x] = h[num_cells[1] * num_cells[0] + x];
137  // Momentum in x direction
138  hu[x] = hu[num_cells[0] + x];
139  hu[(num_cells[1] + 1) * num_cells[0] + x] = hu[num_cells[1] * num_cells[0] + x];
140  // Momentum in y direction
141  hv[x] = reflective_bounds ? -hv[num_cells[0] + x] : hv[num_cells[0] + x];
142  hv[(num_cells[1] + 1) * num_cells[0] + x] = reflective_bounds ? -hv[num_cells[1] * num_cells[0] + x] : hv[
143  num_cells[1] * num_cells[0] + x];
144  }
145 
146  // Pull as much storage as possible out of the loops
147  std::vector<float> h_updates_neg((num_cells[0] + 1) * (num_cells[1] + 2));
148  std::vector<float> h_updates_pos((num_cells[0] + 1) * (num_cells[1] + 2));
149  std::vector<float> hu_updates_neg((num_cells[0] + 1) * (num_cells[1] + 2));
